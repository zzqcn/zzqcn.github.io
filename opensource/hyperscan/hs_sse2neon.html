x86 SSE指令转换为ARM NEON指令¶

本文更新于 2018.05.21

概述¶

Hyperscan使用了intel SIMD指令来进行多项优化动作. 以下摘自hyperscan官方文档 (http://intel.github.io/hyperscan/dev-reference/getting_started.html):

Hyperscan will run on x86 processors in 64-bit (Intel® 64 Architecture) and 32-bit (IA-32 Architecture) modes.

Hyperscan is a high performance software library that takes advantage of recent Intel architecture advances. At a minimum, support for Supplemental Streaming SIMD Extensions 3 (SSSE3) is required, which should be available on any modern x86 processor.

Additionally, Hyperscan can make use of:

  • Intel Streaming SIMD Extensions 4.2 (SSE4.2)
  • the POPCNT instruction
  • Bit Manipulation Instructions (BMI, BMI2)
  • Intel Advanced Vector Extensions 2 (Intel AVX2)

if present.

These can be determined at library compile time, see Target Architecture.

可见我们至少要转换SSSE3指令.

涉及的文件:

文件 说明
config.h configure自动生成,判断CPU型号
util/arch.h 判断指令集, 如SSE4.2, AVX等
util/intrinsics.h 包含编译器提供的头文件, 如x86intrin.h
util/simd_types.h SIMD数据类型定义, 如m128, m256
util/simd_utils.h SIMD指令
util/bitutils.h bit操作原语

ARM NEON指令¶

XXX

SSE to NEON¶

要把hyperscan中的x86指令最小化地移植为NEON指令, 需要做以下几件事:

  • 在make.config定义PLATFORM变量, 指定当前CPU型号, 如x86_64, arm. 日后也可以通过某种方式 自动确定. 还需要将-DPLATFORM=$(PLATFORM)添加到CFLAGS和CXXFLAGS
  • 修改config.h, 根据PLATFROM来定义CPU相关的宏定义
  • 修改arch.h, 根据PLATFORM不同定义不同的条件编译宏
  • 修改intrinsics.h, 对于ARM平台包含SSE2NEON.h头文件
  • 修改simd_types.h, 对ARM平台定义合适的数据类型
  • 个性smid_utils.h和bitutils.h, 进行具体指令的移植

其中, 具体指令的移植是最困难的. 虽然开源代码SSE2NEON.h已经实现了很多SSE到NEON指令的转换, 但还是有一些没有实现, 需要我们自己实现.

具体指令的转换¶

_mm_set_epi64x¶

介绍:

  • 原型

    __m128i _mm_set_epi64x (__int64 e1, __int64 e0)

  • 说明

    Set packed 64-bit integers in dst with the supplied values.

  • 伪码:

    dst[63:0] := e0
dst[127:64] := e1

SSE2NEON已有代码:

FORCE_INLINE __m128i _mm_set_epi32(int i3, int i2, int i1, int i0)
{
    int32_t __attribute__((aligned(16))) data[4] = { i0, i1, i2, i3 };
    return vreinterpretq_m128i_s32(vld1q_s32(data));
}

仿照已有代码添加代码:

FORCE_INLINE __m128i _mm_set_epi64x(int64_t e1, int64_t e0)
{
    int32_t __attribute__((aligned(16))) data[4] =
        { e0 & 0xffffffff,  (e0 >> 32), e1 & 0xffffffff, e1 >> 32 };
    return vreinterpretq_m128i_s32(vld1q_s32(data));
}

_mm_set_epi64x¶

介绍:

  • 原型

    __m128i _mm_set1_epi64x (__int64 a)

  • 说明

    Broadcast 64-bit integer a to all elements of dst. This intrinsic may generate the vpbroadcastq.

  • 伪码:

    FOR j := 0 to 1
    i := j*64
    dst[i+63:i] := a[63:0]
ENDFOR

添加代码:

FORCE_INLINE __m128i _mm_set1_epi64x(int64_t a)
{
    return _mm_set_epi64x(a, a);
}

_mm_slli_epi64¶

  • 原型

    __m128i _mm_slli_epi64 (__m128i a, int imm8)

  • 说明

    Shift packed 64-bit integers in a left by imm8 while shifting in zeros, and store the results in dst.

  • 伪码:

    FOR j := 0 to 1
    i := j*64
    IF imm8[7:0] > 63
        dst[i+63:i] := 0
    ELSE
        dst[i+63:i] := ZeroExtend(a[i+63:i] << imm8[7:0])
    FI
ENDFOR

添加代码:

FORCE_INLINE __m128i _mm_slli_epi64(__m128i a, int imm8)
{
    int64_t *v64val = (int64_t*) &a;
    int64_t r0 = v64val[0] << imm8;
    int64_t r1 = v64val[1] << imm8;
    return _mm_set_epi64x(r1, r0);
}

_mm_sll_epi64¶

  • 原型

    __m128i _mm_sll_epi64 (__m128i a, __m128i count)

  • 说明

    Shift packed 64-bit integers in a left by count while shifting in zeros, and store the results in dst.

  • 伪码:

    FOR j := 0 to 1
    i := j*64
    IF count[63:0] > 63
        dst[i+63:i] := 0
    ELSE
        dst[i+63:i] := ZeroExtend(a[i+63:i] << count[63:0])
    FI
ENDFOR

添加代码:

FORCE_INLINE __m128i _mm_sll_epi64(__m128i a, __m128i count)
{
    int64_t *c64 = (int64_t*) &count;
    int64_t c0 = c64[0];
    int64_t *v64val = (int64_t*) &a;
    int64_t r0 = v64val[0] << c0;
    int64_t r1 = v64val[1] << c0;
    return _mm_set_epi64x(r1, r0);
}

_mm_srli_epi64¶

  • 原型

    __m128i _mm_srli_epi64 (__m128i a, int imm8)

  • 说明

    Shift packed 64-bit integers in a right by imm8 while shifting in zeros, and store the results in dst.

  • 伪码:

    FOR j := 0 to 1
    i := j*64
    IF imm8[7:0] > 63
        dst[i+63:i] := 0
    ELSE
        dst[i+63:i] := ZeroExtend(a[i+63:i] >> imm8[7:0])
    FI
ENDFOR

添加代码:

FORCE_INLINE __m128i _mm_srli_epi64(__m128i a, int imm8)
{
    int64_t *v64val = (int64_t*) &a;
    int64_t r0 = v64val[0] >> imm8;
    int64_t r1 = v64val[1] >> imm8;
    return _mm_set_epi64x(r1, r0);
}

_mm_cmpeq_epi32¶

  • 原型

    __m128i _mm_cmpeq_epi32 (__m128i a, __m128i b)

  • 说明

    Compare packed 32-bit integers in a and b for equality, and store the results in dst.

  • 伪码:

    FOR j := 0 to 3
    i := j*32
    dst[i+31:i] := ( a[i+31:i] == b[i+31:i] ) ? 0xFFFFFFFF : 0
ENDFOR

添加代码:

FORCE_INLINE __m128i _mm_cmpeq_epi32 (__m128i a, __m128i b)
{
    return vreinterpretq_m128i_u32(vceqq_s32(vreinterpretq_s32_m128i(a),
                                   vreinterpretq_s32_m128i(b)));
}

_mm_alignr_epi8¶

  • 原型

    __m128i _mm_alignr_epi8 (__m128i a, __m128i b, int count)

  • 说明

    Concatenate 16-byte blocks in a and b into a 32-byte temporary result, shift the result right by count bytes, and store the low 16 bytes in dst.

  • 伪码:

    tmp[255:0] := ((a[127:0] << 128) OR b[127:0]) >> (count[7:0]*8)
dst[127:0] := tmp[127:0]

添加代码:

FORCE_INLINE __m128i _mm_alignr_epi8(__m128i a, __m128i b, const int count)
{
    switch(count/4)
    {
    case 0:
        return vextq_s32(b, a, 0);
    case 1:
        return vextq_s32(b, a, 1);
    case 2:
        return vextq_s32(b, a, 2);
    case 3:
        return vextq_s32(b, a, 3);
    default:
        assert(0);
        return a;
    }
}

_mm_shuffle_epi8¶

  • 原型

    __m128i _mm_shuffle_epi8 (__m128i a, __m128i b)

  • 说明

    Shuffle packed 8-bit integers in a according to shuffle control mask in the corresponding 8-bit element of b, and store the results in dst.

  • 伪码:

    FOR j := 0 to 15
    i := j*8
    IF b[i+7] == 1
        dst[i+7:i] := 0
    ELSE
        index[3:0] := b[i+3:i]
        dst[i+7:i] := a[index*8+7:index*8]
    FI
ENDFOR

添加代码:

FORCE_INLINE __m128i _mm_shuffle_epi8(__m128i a, __m128i b)
{
    int8_t *a8p = (int8_t*) &a;
    int8_t *b8p = (int8_t*) &b;
    int8_t __attribute__((aligned(16))) ret[16] = {0};
    uint8_t idx;
    int i;

    for(i=0; i<16; i++)
    {
        if(b8p[i] & 0x80)
        {
            ret[i] = 0;
        }
        else
        {
            idx = b8p[i] & 0x0f;
            ret[i] = a8p[idx];
        }
    }

    return vreinterpretq_m128i_s8(vld1q_s8(ret));
}
